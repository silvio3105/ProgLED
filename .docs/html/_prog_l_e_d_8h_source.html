538 void update(ProgLED_line_t type)
539 {
540 // Exit if line is already clocking
541 if (status() == ProgLED_state_t::CLOCKING) return;
542
543 // Calculate new RGB values for each LED
544 ProgLED_LOOP led[i].adjust();
545
546 // Reset LED and bit index
547 ledIdx = 0;
548 ledByte = 0;
549 ledBit = 8;
550
551 // Set new LED line status
552 lineStatus = ProgLED_state_t::CLOCKING;
553
554 // Call external handler
555 if (type == ProgLED_line_t::NON_BLOCKING) startHandler(-1); // Just to fullfill parameter
556 else
557 {
558 uint8_t bit = 0;
559
560 // Fetch first bit
561 fetchBit(bit);
562
563 // Call external handler with first bit to clock out
564 startHandler(bit);
565 }
566 }

592 uint8_t fetchBit(uint8_t& bit)
593 {
594 // Get bit from color output data
595 bit = SSTD_BBIT(led[ledIdx].getColor(ledByte), (ledBit - 1));
596
597 // Set bit to 0 if LED status is 0(OFF)
598 if (!(SSTD_BIT(led[ledIdx].getConfig(), PROG_LED_STATUS_BIT))) bit = 0;
599
600 // Move bit index
601 ledBit--;
602
603 // If end of byte is reached
604 if (!ledBit)
605 {
606 // Reset bit index to 0
607 ledBit = 8;
608
609 // Move index to next byte
610 ledByte++;
611
612 // If end of LED is reached
613 if (ledByte == 3)
614 {
615 // Move index to next LED
616 ledByte = 0;
617 ledIdx++;
618
619 // If end is reached return OK status
620 if (ledIdx == ledNum) return PROG_LED_OK;
621 }
622 }
623
624 // Return continue status
625 return PROG_LED_CONTINUE;
626 }

639 uint8_t fillBuffer(uint8_t* buff, uint32_t len, uint32_t qty, uint8_t low, uint8_t high)
640 {
641 uint8_t bit = 0;
642 uint8_t ret = PROG_LED_OK;
643 uint32_t idx = 0;
644
645 // Increase qty by 24 bits
646 qty *= 24;
647
648 // Buffer is smaller than required by qty parameter
649 if (len < qty) return PROG_LED_NOK;
650
651 // Fill buffer while continue flag is received and qty is not 0
652 do
653 {
654 // Fetch bit
655 ret = fetchBit(bit);
656
657 // Write high or low value
658 if (bit) buff[idx] = high;
659 else buff[idx] = low;
660
661 // Move index
662 idx++;
663
664 // Decrease qty
665 qty--;
666 }
667 while (ret != PROG_LED_OK && qty);
668
669 // Return OK status
670 return PROG_LED_OK;
671 }
